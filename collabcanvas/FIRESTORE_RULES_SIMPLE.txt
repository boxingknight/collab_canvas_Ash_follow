rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Shapes collection
    match /shapes/{shapeId} {
      // Allow authenticated users to read all shapes
      allow read: if request.auth != null;
      
      // Allow authenticated users to create shapes
      // Ensure createdBy matches the user's UID
      allow create: if request.auth != null 
                    && request.resource.data.createdBy == request.auth.uid;
      
      // Allow authenticated users to update any shape (collaborative editing)
      allow update: if request.auth != null;
      
      // Allow authenticated users to delete any shape (for MVP simplicity)
      allow delete: if request.auth != null;
    }
    
    // Cursors collection (for multiplayer cursor tracking)
    match /cursors/{userId} {
      // Allow authenticated users to read all cursors
      allow read: if request.auth != null;
      
      // Allow users to write/update/delete only their own cursor
      allow write: if request.auth != null 
                   && request.auth.uid == userId;
    }
    
    // Presence collection (for online user tracking)
    match /presence/{userId} {
      // Allow authenticated users to read all presence data
      allow read: if request.auth != null;
      
      // Allow users to write/update/delete only their own presence
      allow write: if request.auth != null 
                   && request.auth.uid == userId;
    }
  }
}
